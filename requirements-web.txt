# Slim dependency set for the production web API (loom_api.web).
#
# Scope: only what POST /romanize, POST /annotate, GET /language/config/{code},
# and GET /health actually reach.  No video / no Playwright / no Streamlit /
# no OCR / no Pillow / no fonttools / no ffmpeg.
#
# Used by: Railway deploy at api.loom.nerv-analytic.ai.  Local dev devs
# install requirements.txt instead so they get the full toolchain.
#
# When adding a new language to loom_core/romanize.py, add its pip dep here
# too — otherwise Railway will 500 on the first request that exercises it.

# ── Web framework ──────────────────────────────────────────────────────
fastapi
uvicorn[standard]
gunicorn                               # master process for worker recycling (ROMANIZATION_CACHE.md §4 Option A)
uvicorn-worker                         # gunicorn worker class (uvicorn.workers is deprecated upstream)
pydantic>=2.0
slowapi                                # IP rate limiting (4f hardening)

# ── Result cache (ROMANIZATION_CACHE.md Layer 1) ───────────────────────
# Postgres-backed content-addressed cache for /romanize/batch +
# /annotate/batch.  Fail-open: without DATABASE_URL (or with the DB down)
# the API runs uncached.  Web-only deps — desktop requirements.txt omits
# them on purpose (loom_api/result_cache.py imports psycopg lazily).
psycopg[binary]
psycopg-pool

# ── Core dependencies ──────────────────────────────────────────────────
pysubs2                                # imported by loom_core.models / language
langdetect                             # _dominant_script fallback
langcodes
language_data                          # transitive of langcodes

# ── Per-language romanization deps ─────────────────────────────────────
# Japanese — fugashi (MeCab) + unidic-lite dictionary
fugashi
unidic-lite

# Chinese — pinyin / zhuyin / Traditional→Simplified
pypinyin
jieba
opencc-python-reimplemented

# Cantonese — jyutping
pycantonese

# Korean — Revised Romanization
korean-romanizer

# Cyrillic — ru / uk / be / sr / bg / mk / mn
cyrtranslit

# Thai — paiboon / RTGS / IPA (RTGS engine = thai2rom = neural, needs torch)
pythainlp
python-crfsuite                        # han_solo CRF segmenter (transitive)
numpy                                  # thai2rom calls .numpy() on torch tensors
# Pin torch to the CPU-only PyPI wheel — the default index pulls CUDA
# builds (~800MB).  Railway gets the ~700MB CPU build via this hint
# (still much smaller than the ~2GB CUDA install).
--extra-index-url https://download.pytorch.org/whl/cpu
torch

# Indic — Devanagari / Bengali / Tamil / Telugu / Gujarati / Gurmukhi
aksharamukha
